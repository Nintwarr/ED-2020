51  // podemos saber si estos pertenecen a un extremo del

98 bool umbralizado_automatico(const char *fichero_entrada, const char *fichero_salida, byte &umbral) {
99  bool exito = false;
100  bool primera_vez = true;
101  Imagen *imagen = new Imagen();
102  double umbral_actual = 0, umbral_siguiente = 0;
103  double umbral_primero = 0, umbral_segundo = 0;
104  double epsilon = 0.00001;
105 
106  // Si podemos cargar la imagen seguimos
107  if (imagen->cargar_imagen_PGM(fichero_entrada)) {
108  exito = true;
109 
110  // La primera vez que entramos al cálculo de la media de
111  // umbrales, debemos usar como referencia la media de
112  // la imagen leída de fichero. Las siguientes veces tenemos
113  // que tomar como referencia el cálculo conseguido en la iteración anterior
114  do {
115  if (primera_vez) {
116  umbral_actual = calcular_media_gris(imagen);
117  primera_vez = false;
118  }
119  else {
120  umbral_actual = umbral_siguiente;
121  }
122 
123  // Se calculan las medias de grises sobre la imagen cargada
124  // separando la selección de píxeles en función del umbral
125  // o media de tonos de gris que ya hemos calculado
126  separar_segun_umbral(imagen,umbral_actual,umbral_primero,umbral_segundo);
127  // Con esas dos medias, se vuelve a hacer una media
128  umbral_siguiente = (umbral_primero + umbral_segundo)/2;
129  // Ciclamos mientras el valor absoluto de la diferencia
130  // entre el umbral calculado tras la separación y el de la
131  // iteración anterior sea mayor o igual que un valor
132  // muy pequeño. Es decir, buscamos que la diferencia
133  // entre la media de una iteración y otra, sea muy
134  // pequeña
135  } while (abs(umbral_siguiente-umbral_actual) >= epsilon);
136 
137  // Como en la imagen se guardan bytes, eso es lo que el usuario
138  // espera que use como umbral. Así que redondeamos el umbral,
139  // lo convertimos a byte y lo asignamos al parámetro de salida
140  umbral = byte(round(umbral_siguiente));
141 
142  // Hacemos efectiva la umbralización con el umbral calculado
143  umbralizar_basico(imagen,umbral);
144 
145  // Si no podemos escribir la imagen, entendemos que hemos
146  // fracasado
147  exito = imagen->escribir_imagen_PGM(fichero_salida);
148  }
149 
150  delete imagen;
151 
152  return exito;
153 }
